SELECT PPCI.ID_ITM SKU,
	    AI.DE_ITM PRODUCTO,
	    E.ID_STR_RT TIENDA,
	    PA.NM_LOC FARMACIA,
	    SUBSTR (AI.ID_MRHRC_GP, 3, 2) CATEGORIA,
	    PPCI.MO_OVRD_PRC PRECIO,
	    SUBSTR (TS_EV_MNT_EF, 1, 8) FECHA,
	    SUBSTR (TS_EV_MNT_EF, 10, 20) HOUR,
	    E.ID_EV ID_EVENTO,
	    ID_EV_EXT
		--TY_EV_MNT,
		--mo_chn_prn_un_prc,
		--ty_chn_prn_un_prc
FROM CO_EV E
	JOIN CO_EV_MNT EM ON EM.ID_EV = E.ID_EV
	AND
		EM.ID_STR_RT = E.ID_STR_RT
	JOIN MA_PRC_ITM MPI ON MPI.ID_EV = EM.ID_EV
	AND
		MPI.ID_STR_RT = EM.ID_STR_RT
	JOIN TR_CHN_PRN_PRC PPC ON PPC.ID_EV = MPI.ID_EV
	AND
		PPC.ID_STR_RT = MPI.ID_STR_RT
	JOIN MA_ITM_PRN_PRC_ITM PPCI ON PPCI.ID_EV = PPC.ID_EV
	AND
		PPCI.ID_STR_RT = PPC.ID_STR_RT
	JOIN AS_ITM AI ON AI.ID_ITM = PPCI.ID_ITM
	JOIN PA_STR_RTL PA ON PA.ID_STR_RT = PPC.ID_STR_RT
WHERE PPCI.ID_ITM = 100806
ORDER BY 6;
